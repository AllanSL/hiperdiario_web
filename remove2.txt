        return Array.from(new Set(catalog.map(m => m.active_principle))).sort();
    }, [catalog]);

    const availableStrengths = useMemo(() => {
        if (!selectedPrinciple) return [];
        const strengths = catalog
            .filter(m => m.active_principle === selectedPrinciple)
            .map(m => m.strength);
        return Array.from(new Set(strengths)).sort();
    }, [catalog, selectedPrinciple]);

    const selectedMedObj = useMemo(() => {
        if (!selectedPrinciple || !selectedStrength) return null;
        return catalog.find(m => m.active_principle === selectedPrinciple && m.strength === selectedStrength) || null;
    }, [selectedPrinciple, selectedStrength, catalog]);

    const handleSubmit = async (e: React.FormEvent) => {
        e.preventDefault();
        if (!selectedMedObj) {
            setMsg({ text: 'Selecione um medicamento válido do catálogo.', type: 'error' });
            return;
        }

        setLoading(true);
        setMsg({ text: '', type: '' });

        try {
            const { error } = await supabase.from('pharmacy_inventory').upsert({
                catalog_id: selectedMedObj.id,
                ubs_cnes: cnes,
                quantity_in_stock: Number(initialStock) || 0,
                last_updated_at: new Date().toISOString()
            }, {
                onConflict: 'catalog_id,ubs_cnes'
            });

            if (error) throw error;

            setMsg({ text: 'Medicamento adicionado ao estoque local com sucesso!', type: 'success' });
            setSelectedPrinciple('');
            setSelectedStrength('');
            setInitialStock('');
            setTimeout(() => window.location.reload(), 2000);
        } catch (err: any) {
            setMsg({ text: err.message || 'Erro ao adicionar.', type: 'error' });
        } finally {
            setLoading(false);
        }
    };

    return (
        <div className="bg-white p-6 rounded-xl shadow-sm border border-gray-100">
            <div className="flex items-center gap-3 mb-6">
                <div className="p-3 bg-teal-50 text-teal-600 rounded-lg"><PackageSearch size={24} /></div>
                <h3 className="text-lg font-semibold">Adicionar Medicamento ao Estoque Local</h3>
            </div>
            
            <p className="text-sm text-gray-500 mb-6">
                Selecione o medicamento no banco padrão do SUS para adicionar ao estoque da sua unidade de saúde.
            </p>

            <form onSubmit={handleSubmit} className="max-w-2xl space-y-6">
                {msg.text && (
                    <div className={`p-3 rounded-lg text-sm font-medium ${msg.type === 'success' ? 'bg-green-50 text-green-700' : 'bg-red-50 text-red-700'}`}>
                        {msg.text}
                    </div>
                )}

                <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
                    <div className="relative z-20">
                        <label className="block text-sm font-medium text-gray-700 mb-1">Princípio Ativo *</label>
                        <CustomSelect 
                            options={uniquePrinciples.map(p => ({ value: p, label: p }))} 
                            value={selectedPrinciple} 
                            onChange={(val) => {
                                setSelectedPrinciple(val as string);
                                setSelectedStrength('');
                            }} 
                            placeholder="Selecione o medicamento" 
                        />
                    </div>
                    
                    <div className="relative z-10">
                        <label className="block text-sm font-medium text-gray-700 mb-1">Dosagem / Concentração *</label>
                        <CustomSelect 
                            disabled={!selectedPrinciple} 
                            options={availableStrengths.map(s => {
                                const m = catalog.find(med => med.active_principle === selectedPrinciple && med.strength === s);
                                return { value: s, label: `${s} (${m?.form})` };
                            })} 
                            value={selectedStrength} 
                            onChange={(val) => setSelectedStrength(val as string)} 
                            placeholder="Selecione" 
                        />
                    </div>
                </div>

                {selectedMedObj && (
                    <div className="mt-4 p-4 border border-teal-200 bg-teal-50 rounded-lg">
                        <div className="mb-4">
                            <p className="font-semibold text-teal-800">
                                {selectedMedObj.active_principle} {selectedMedObj.strength}
                            </p>
                            <p className="text-sm text-teal-600">{selectedMedObj.category} - {selectedMedObj.form}</p>
                        </div>
                        
                        <div>
                            <label className="block text-sm font-medium text-teal-800 mb-1">Estoque Inicial</label>
                            <div className="flex items-center gap-2">
                                <input 
                                    required 
                                    type="number" 
                                    min="0" 
                                    className="w-48 border border-gray-300 rounded-lg p-2.5 text-sm focus:ring-1 focus:ring-teal-500 outline-none" 
                                    value={initialStock} 
                                    onChange={e => setInitialStock(e.target.value ? Number(e.target.value) : '')} 
                                />
                                <span className="text-sm text-teal-700 font-medium">{selectedMedObj.dispensing_unit}s</span>
                            </div>
                        </div>
                    </div>
                )}

                <div className="pt-4 border-t border-gray-100 flex justify-end">
                    <button type="submit" disabled={loading || !selectedMedObj} className="bg-teal-600 hover:bg-teal-700 text-white font-medium px-6 py-2.5 rounded-lg shadow-sm transition disabled:opacity-50">
                        {loading ? 'Adicionando...' : 'Adicionar à UBS'}
                    </button>
                </div>
            </form>
        </div>
    );
}
